DROP TABLE DELIVERY_LIST;

CREATE TABLE DELIVERY_LIST(
    ID     NUMBER(4) CONSTRAINT DEL_ID_PK PRIMARY KEY,
    SUPPLIER VARCHAR2(20) CONSTRAINT DEL_SUP_NN NOT NULL,
    DEL_DATE DATE CONSTRAINT DEL_DT_NN NOT NULL,
    SIZE_500 NUMBER(5),
    SIZE_1000 NUMBER(5)
);

DROP SEQUENCE DELIVERY_SEQ;

CREATE SEQUENCE DELIVERY_SEQ
    INCREMENT BY 1
    START WITH 1
    MINVALUE 1
    MAXVALUE 9999;
